## Assignment 3.1 - 分支逻辑对比

### 32 位版本

```asm
your_if:
    mov eax, dword [a1]      ; eax = a1
    cmp eax, 12
    jge elif_2
    
    ; 分支 1：if_flag = a1/2 + 1
    mov eax, dword [a1]
    mov ecx, 2
    cdq                       ; 符号扩展
    idiv ecx
    add eax, 1
    mov dword [if_flag], eax
    jmp if_done
    
elif_2:
    mov eax, dword [a1]
    cmp eax, 24
    jge else_3
    
    ; 分支 2：if_flag = (24 - a1) * a1
    mov eax, 24
    mov ecx, dword [a1]
    sub eax, ecx
    imul eax, ecx
    mov dword [if_flag], eax
    jmp if_done
    
else_3:
    ; 分支 3：if_flag = a1 << 4
    mov eax, dword [a1]
    shl eax, 4
    mov dword [if_flag], eax
    
if_done:
```

### 64 位版本

```asm
your_if:
    mov eax, dword [rel a1]   ; 改动 1：加 [rel] 相对寻址
    cmp eax, 12
    jge elif_2
    
    ; 分支 1：if_flag = a1/2 + 1
    mov eax, dword [rel a1]    ; 改动 1：加 [rel]
    mov ecx, 2
    cdq                        ; 保持不变（处理 32 位）
    idiv ecx
    add eax, 1
    mov dword [rel if_flag], eax ; 改动 1：加 [rel]
    jmp if_done
    
elif_2:
    mov eax, dword [rel a1]    ; 改动 1：加 [rel]
    cmp eax, 24
    jge else_3
    
    ; 分支 2：if_flag = (24 - a1) * a1
    mov eax, 24
    mov ecx, dword [rel a1]    ; 改动 1：加 [rel]
    sub eax, ecx
    imul eax, ecx
    mov dword [rel if_flag], eax ; 改动 1：加 [rel]
    jmp if_done
    
else_3:
    ; 分支 3：if_flag = a1 << 4
    mov eax, dword [rel a1]    ; 改动 1：加 [rel]
    shl eax, 4
    mov dword [rel if_flag], eax ; 改动 1：加 [rel]
    
if_done:
```

**改动总结**：只需在所有内存访问前加 `rel`（RIP 相对寻址）

## Assignment 3.2 - 循环逻辑对比

### 32 位版本

```asm
your_while:
    mov ecx, dword [a2]      ; ecx = a2（计数器）
    
while_loop:
    cmp ecx, 12
    jl while_end
    
    call my_random           ; 返回值在 eax
    
    mov ebx, ecx
    sub ebx, 12
    mov byte [while_flag + ebx], al   ; 存储字符
    
    dec ecx
    jmp while_loop
    
while_end:
    mov dword [a2], ecx      ; 更新 a2
```

### 64 位版本

```asm
your_while:
    mov r8d, dword [rel a2]  ; 改动 1：加 rel，用 r8d 代替 ecx
    
while_loop:
    cmp r8d, 12              ; 改动 2：ecx → r8d
    jl while_end
    
    call my_random           ; 返回值在 eax（保持一致）
    
    mov ecx, r8d             ; 改动 3：改为 r8d 赋值给 ecx
    sub ecx, 12
    mov byte [rel while_flag + rcx], al   ; 改动 4：加 rel，rcx 用 64 位
    
    dec r8d                  ; 改动 5：ecx → r8d
    jmp while_loop
    
while_end:
    mov dword [rel a2], r8d  ; 改动 6：加 rel，ecx → r8d
```

**改动总结**：
- 加 `rel` 进行 RIP 相对寻址
- `ecx` → `r8d`（用新的 64 位寄存器）
- `[while_flag + ebx]` → `[rel while_flag + rcx]`（地址计算用 64 位寄存器）

## Assignment 3.3 - 函数调用对比

### 32 位版本

```asm
your_function:
    push ebp
    mov ebp, esp
    
    xor ecx, ecx              ; ecx = 0（计数器）
    
loop_start:
    mov al, byte [your_string + ecx]  ; al = string[i]
    cmp al, 0
    je loop_end
    
    pushad                     ; 保存所有寄存器
    
    push eax                   ; 参数入栈
    call print_a_char          ; 调用（参数从栈读）
    add esp, 4                 ; 清理栈
    
    popad                      ; 恢复所有寄存器
    
    inc ecx
    jmp loop_start
    
loop_end:
    pop ebp
    ret
```

### 64 位版本

```asm
your_function:
    push rbp                   ; 改动 1：ebp → rbp
    mov rbp, rsp               ; 改动 2：esp → rsp
    
    xor r8d, r8d               ; 改动 3：ecx → r8d（新寄存器）
    
loop_start:
    mov rsi, [rel your_string] ; 改动 4：加 rel，用 rsi 读地址
    mov al, byte [rsi + r8]    ; 改动 5：your_string → rsi，ecx → r8
    cmp al, 0
    je loop_end
    
    ; 改动 6：没有 pushad/popad（64 位不支持）
    ; 改动 7：不需要手动保存/恢复（caller saved）
    
    movzx edi, al              ; 改动 8：参数通过 rdi，不是栈
    call print_a_char          ; 调用（参数从 rdi 读）
    
    ; 改动 9：不需要清理栈
    
    inc r8d                    ; 改动 10：ecx → r8d
    jmp loop_start
    
loop_end:
    pop rbp                    ; 改动 11：ebp → rbp
    ret
```

**改动总结**：
- 寄存器：所有 32 位改 64 位
- 参数传递：从栈改寄存器（第 1 参数在 rdi）
- 消除 `pushad/popad`
- 加 `rel` 进行 RIP 相对寻址

## 快速改动清单

按顺序做这些改动就能从 32 位升级到 64 位：

### 第 1 步：寄存器名称

| 旧 (32位) | 新 (64位) |
|----------|----------|
| eax | rax / eax（如果只用 32 位值） |
| ebx | rbx / ebx |
| ecx | rcx / ecx / r8d / r8（根据需要） |
| edx | rdx / edx |
| esp | rsp |
| ebp | rbp |
| esi | rsi |
| edi | rdi |

### 第 2 步：内存寻址

```
[a1]                → [rel a1]
[eax + ebx*4]       → [rax + rbx*8]（注意大小也许变化）
[esp + 4]           → [rsp + 8]（栈大小从 4→8）
```

### 第 3 步：调用约定

```
32 位：push 参数
64 位：mov rdi, 参数1; mov rsi, 参数2; ...

32 位：pop eax（清理）
64 位：无需清理
```

### 第 4 步：移除 pushad/popad

要么直接移除，要么改成：
```asm
push rax
push rbx
push rcx
; ...
pop rcx
pop rbx
pop rax
```

### 第 5 步：编译命令改动

```bash
# 32 位
nasm -f elf32 student.asm -o student.o
gcc -m32 -c test.cpp
gcc -m32 -o test student.o test.o

# 64 位
nasm -f elf64 student.asm -o student.o
gcc -m64 -c test.cpp
gcc -m64 -o test student.o test.o
```

## 一行总结对比表

| 操作 | 32 位代码 | 64 位代码 |
|------|---------|---------|
| **编译** | `nasm -f elf32` | `nasm -f elf64` |
| **寄存器** | eax, ecx, ... | rax, rcx, ...（或 r8-r15） |
| **内存读** | `mov eax, [a1]` | `mov eax, [rel a1]` |
| **参数传递** | `push arg; call` | `mov rdi, arg; call` |
| **返回值** | eax | rax |
| **保存寄存器** | pushad/popad | push/pop 个别 |
| **符号扩展** | cdq | cqo（或 cdq for 32bit） |
| **栈大小** | 4 字节 | 8 字节 |

## 常见错误速查

**错误 1**：`[a1]` 在 64 位不能工作

```asm
❌ mov eax, [a1]
✅ mov eax, [rel a1]
```

**错误 2**：忘记改寄存器宽度

```asm
❌ mov eax, dword [rsi + rcx*4]  ; 混用了 eax 和 rsi/rcx
✅ mov rax, qword [rsi + rcx*8]
✅ mov eax, dword [rsi + rcx*4]  ; 如果确实只用 32 位
```

**错误 3**：调用 C++ 函数参数还在栈上

```asm
32 位: ✅ push eax; call func; add esp, 4
64 位: ❌ push eax; call func; add rsp, 8
64 位: ✅ mov rdi, eax; call func
```

**错误 4**：使用 pushad/popad

```asm
32 位: ✅ pushad; ... ; popad
64 位: ❌ pushad; popad（不存在！）
64 位: ✅ push rax; push rbx; ...; pop rbx; pop rax
```

如果你想试试，将以下代码保存为 `student_64.asm`：

```asm
bits 64

extern a1, a2, my_random, your_string, if_flag, while_flag, print_a_char

section .text
global student_function, your_function

student_function:
    ret

your_if:
    mov eax, dword [rel a1]
    cmp eax, 12
    jge elif_2
    mov eax, dword [rel a1]
    mov ecx, 2
    cdq
    idiv ecx
    add eax, 1
    mov dword [rel if_flag], eax
    jmp if_done
elif_2:
    mov eax, dword [rel a1]
    cmp eax, 24
    jge else_3
    mov eax, 24
    mov ecx, dword [rel a1]
    sub eax, ecx
    imul eax, ecx
    mov dword [rel if_flag], eax
    jmp if_done
else_3:
    mov eax, dword [rel a1]

    mov dword [rel if_flag], eax
if_done:
    ret

your_while:
    mov r8d, dword [rel a2]
while_loop:
    cmp r8d, 12
    jl while_end
    call my_random
    mov ecx, r8d
    sub ecx, 12

    ret
```

