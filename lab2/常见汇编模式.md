# 汇编常见模式速查手册

> 快速查找常用的汇编代码片段

---

## 模式 1：简单赋值

### C++ 代码
```cpp
int x = 10;
int y = x + 5;
```

### 汇编
```asm
mov eax, 10          ; eax = 10
add eax, 5           ; eax = eax + 5
mov [y], eax         ; 存到 y
```

## 模式 2：数组访问

### C++ 代码
```cpp
int arr[5] = {1, 2, 3, 4, 5};
int value = arr[2];  // 访问第 3 个元素
arr[0] = 99;         // 修改第 1 个元素
```

### 汇编
```asm
; 假设 arr 的地址在 esi

; 读 arr[2]
mov eax, [esi + 2*4]    ; 每个元素 4 字节，所以偏移 2*4 = 8

; 写 arr[0] = 99
mov dword [esi], 99     ; 直接写到首地址
```

**规律**：`arr[i] = *(arr + i*元素大小)`

## 模式 3：条件判断（单 if）

### C++ 代码
```cpp
if (x > 10) {
    y = 1;
}
```

### 汇编
```asm
mov eax, [x]         ; eax = x
cmp eax, 10          ; 比较
jle skip              ; 如果 <=10，跳过

; 执行 if 内容
mov dword [y], 1

skip:
```

## 模式 4：条件判断（if-else）

### C++ 代码
```cpp
if (x < 5) {
    y = 100;
} else {
    y = 200;
}
```

### 汇编
```asm
mov eax, [x]
cmp eax, 5
jge else_part        ; >= 5 跳到 else

; if 部分
mov dword [y], 100
jmp if_end

else_part:
; else 部分
mov dword [y], 200

if_end:
```

## 模式 5：while 循环

### C++ 代码
```cpp
int i = 0;
while (i < 10) {
    sum += arr[i];
    i++;
}
```

### 汇编
```asm
xor ecx, ecx         ; ecx = 0（i = 0）

while_loop:
    cmp ecx, 10      ; i < 10 ?
    jge loop_end     ; >= 10 退出

    ; 循环体：sum += arr[i]
    mov eax, [arr + ecx*4]
    add [sum], eax

    inc ecx           ; i++
    jmp while_loop

loop_end:
```

    mov eax, [arr + ecx*4]

for_end:

## 模式 7：嵌套循环

### C++ 代码
```cpp
for (int i = 0; i < 3; i++) {
    for (int j = 0; j < 4; j++) {
        matrix[i][j] = 0;
    }
}
```

### 汇编
```asm
xor ecx, ecx         ; i = 0

outer_loop:
    cmp ecx, 3
    jge outer_end

    xor edx, edx     ; j = 0 （内循环，用不同寄存器！）

    inner_loop:
        cmp edx, 4
        jge inner_end

        ; matrix[i*4 + j] = 0
        mov eax, ecx
        imul eax, 4           ; eax = i * 4
        add eax, edx          ; eax = i*4 + j
        mov dword [matrix + eax*4], 0

        inc edx
        jmp inner_loop

    inner_end:
    inc ecx
    jmp outer_loop

outer_end:
```

**规律**：嵌套循环使用**不同的寄存器**（i 用 ecx，j 用 edx）

## 模式 8：函数调用

### C++ 代码
```cpp
int result = add(3, 5);
```

### 汇编
```asm
; cdecl 调用约定（参数从右到左）
push 5               ; 第 2 个参数
push 3               ; 第 1 个参数
call add             ; 调用函数，返回值在 eax
add esp, 8           ; 清理栈（2 个参数 × 4 字节）

mov [result], eax    ; 保存返回值
```

## 模式 9：简单函数实现

### C++ 代码
```cpp
int double_it(int x) {
    return x * 2;
}
```

### 汇编
```asm
double_it:
    push ebp
    mov ebp, esp

    ; ebp+8 是参数 x
    mov eax, [ebp+8]
    add eax, eax     ; eax = eax + eax（= eax * 2）

    pop ebp
    ret
```

## 模式 10：保存和恢复寄存器

### C++ 代码
```cpp
void modify() {
    int save_eax = eax;  // 保存
    eax = 999;           // 修改
    // 恢复
    eax = save_eax;
}
```

### 汇编方法 1：使用栈
```asm
push eax             ; 保存 eax
mov eax, 999         ; 修改
pop eax              ; 恢复
```

### 汇编方法 2：使用 pushad/popad
```asm
pushad               ; 保存所有寄存器
mov eax, 999
mov ebx, 111
add ecx, 50
popad                ; 恢复所有寄存器
```

## 模式 11：访问 C++ 全局变量

### C++ 代码（在 test.cpp 中）
```cpp
extern int a1;
extern int if_flag;
extern char* your_string;
```

### 汇编代码
```asm
extern a1           ; 声明外部变量
extern if_flag
extern your_string

your_code:
    mov eax, [a1]           ; 读取 a1 的值
    add eax, 10
    mov [if_flag], eax      ; 写入 if_flag

    mov esi, [your_string]  ; 读取字符串首地址
    mov al, [esi]           ; 读取第一个字符
```

## 模式 12：调用 C++ 函数

### C++ 代码（在 test.cpp 中）
```cpp
extern "C" char my_random();
extern "C" void print_a_char(char c);
```

### 汇编代码
```asm
extern my_random       ; 声明外部函数
extern print_a_char

your_code:
    ; 调用 my_random()，返回值在 eax
    call my_random
    mov [result], al    ; al 是返回值

    ; 调用 print_a_char('A')
    push 'A'            ; 参数
    call print_a_char
    add esp, 4          ; 清理栈
```

## 模式 13：乘法运算

### C++ 代码
```cpp
int result = a * b;
```

### 汇编
```asm
mov eax, [a]         ; eax = a
mov ecx, [b]         ; ecx = b
imul eax, ecx        ; eax = eax * ecx
mov [result], eax
```

**注意**：乘法的两个操作数**不能都是内存地址**

## 模式 14：除法运算

### C++ 代码
```cpp
int quotient = a / b;
int remainder = a % b;
```

### 汇编
```asm
mov eax, [a]         ; eax = 被除数
mov ecx, [b]         ; ecx = 除数
cdq                  ; 符号扩展（重要！）
idiv ecx             ; eax = 商，edx = 余数
mov [quotient], eax
mov [remainder], edx
```

**记住**：除法前必须 `cdq`！

## 模式 15：位移运算

### C++ 代码
```cpp
int x = a << 4;  // 左移 4 位（×16）
int y = b >> 2;  // 右移 2 位（÷4）
```

### 汇编
```asm
mov eax, [a]
shl eax, 4           ; eax <<= 4
mov [x], eax

mov eax, [b]
shr eax, 2           ; eax >>= 2
mov [y], eax
```

## 模式 16：字符串遍历

### C++ 代码
```cpp
for (int i = 0; your_string[i] != '\0'; i++) {
    char c = your_string[i];
    print_a_char(c);
}
```

### 汇编
```asm
xor ecx, ecx         ; i = 0
mov esi, [your_string]  ; esi = 字符串首地址

loop:
    mov al, [esi + ecx]  ; al = your_string[i]
    cmp al, 0            ; 是否为 '\0'
    je loop_end

    ; 打印字符
    pushad
    push eax
    call print_a_char
    add esp, 4
    popad

    inc ecx
    jmp loop

loop_end:
```

## 模式 17：数组求和

### C++ 代码
```cpp
int sum = 0;
for (int i = 0; i < 10; i++) {
    sum += arr[i];
}
```

### 汇编
```asm
xor eax, eax         ; eax = 0（sum）
xor ecx, ecx         ; ecx = 0（i）
mov esi, [arr]       ; esi = 数组首地址

loop:
    cmp ecx, 10
    jge loop_end

    add eax, [esi + ecx*4]  ; sum += arr[i]
    inc ecx
    jmp loop

loop_end:
mov [sum], eax
```

## 模式 18：条件的复杂组合

### C++ 代码
```cpp
if (a > 5 && b < 10) {
    result = 1;
} else {
    result = 0;
}
```

### 汇编
```asm
xor eax, eax         ; result = 0（默认）

mov ecx, [a]
cmp ecx, 5
jle end_if           ; a <= 5，则跳过

mov ecx, [b]
cmp ecx, 10
jge end_if           ; b >= 10，则跳过

; 两个条件都满足
mov eax, 1

end_if:
mov [result], eax
```

## 记住这些规则

| 规则 | 说明 |
|------|------|
| **push/pop 配对** | 每个 push 必须对应 pop |
| **cdq 在除法前** | 除法前必须做 cdq |
| **返回值在 eax** | 函数返回值放在 eax（或 edx:eax） |
| **参数从右到左** | cdecl 约定，最右边的参数先 push |
| **不能内存→内存** | 必须借助寄存器 |
| **嵌套循环用不同寄存器** | i 用 ecx，j 用 edx |
| **cmp 后接跳转** | cmp 设置标志位，然后立即跳转 |

## 快速调试清单

- [ ] 所有 `push` 都有对应的 `pop`？
- [ ] 所有 `call` 都有 `ret`？
- [ ] 所有除法前都有 `cdq`？
- [ ] 栈对齐了吗？（push/pop 数量相等）
- [ ] 寄存器有无冲突？（嵌套时是否用了同一个寄存器）
- [ ] 内存地址计算正确吗？
- [ ] 外部变量和函数都 `extern` 了吗？
